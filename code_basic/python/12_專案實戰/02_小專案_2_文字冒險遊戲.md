# 小專案 2：文字冒險遊戲

**目標**: 練習流程控制、巢狀條件判斷以及故事邏輯的設計。

## 程式碼實作

```python
import time

def print_slow(text):
    """模擬打字效果"""
    print(text)
    time.sleep(1)

def intro():
    print_slow("你醒來時，發現自己身處在一個黑暗的森林中。")
    print_slow("眼前有兩條路，一條向左，一條向右。")

def left_path():
    print_slow("\n你選擇了向左走...")
    print_slow("你遇到了一隻飢餓的熊！")
    choice = input("你要 (1) 逃跑 還是 (2) 戰鬥？ ")
    
    if choice == '1':
        print_slow("你轉身就跑，幸運地逃脫了。")
        print_slow("你回到了起點。")
        intro() # 簡化處理：回到開頭
    elif choice == '2':
        print_slow("你勇敢地拔出樹枝與熊戰鬥...")
        print_slow("但熊實在太強大了，你輸了。遊戲結束。")
    else:
        print_slow("你猶豫不決，被熊抓住了。遊戲結束。")

def right_path():
    print_slow("\n你選擇了向右走...")
    print_slow("你發現了一個神秘的寶箱。")
    choice = input("你要 (1) 打開它 還是 (2) 無視它？ ")

    if choice == '1':
        print_slow("寶箱裡裝滿了金幣！你發財了！")
        print_slow("恭喜你通關！")
    elif choice == '2':
        print_slow("你決定不冒險，繼續往前走，最終走出了森林。")
        print_slow("安全通關。")
    else:
        print_slow("你在寶箱前發呆太久，天黑了。遊戲結束。")

def start_game():
    print("=== 森林冒險 ===")
    intro()
    
    while True:
        choice = input("\n請問你要往哪邊走？(左/右): ")
        if choice == "左":
            left_path()
            break
        elif choice == "右":
            right_path()
            break
        else:
            print("請輸入 '左' 或 '右'。")

if __name__ == "__main__":
    start_game()
```
