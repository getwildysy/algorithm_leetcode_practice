# 小專案 3：檔案整理工具

**目標**: 練習 `os` 和 `shutil` 模組的使用，操作檔案系統、移動檔案。

**功能**: 將當前目錄下的檔案，依照副檔名自動分類到不同的資料夾中（如 `Images`, `Documents`, `Scripts`）。

## 程式碼實作

```python
import os
import shutil

# 定義副檔名與目標資料夾的對應
EXTENSIONS = {
    'Images': ['.jpg', '.jpeg', '.png', '.gif', '.bmp'],
    'Documents': ['.pdf', '.docx', '.txt', '.xlsx', '.pptx'],
    'Scripts': ['.py', '.js', '.sh', '.bat'],
    'Archives': ['.zip', '.rar', '.tar.gz']
}

def organize_files(directory):
    # 確保目標路徑存在
    if not os.path.exists(directory):
        print(f"路徑不存在: {directory}")
        return

    # 遍歷目錄中的所有檔案
    for filename in os.listdir(directory):
        file_path = os.path.join(directory, filename)

        # 跳過資料夾
        if os.path.isdir(file_path):
            continue
        
        # 取得副檔名 (轉為小寫)
        _, ext = os.path.splitext(filename)
        ext = ext.lower()

        # 判斷應該移動到哪個資料夾
        moved = False
        for folder_name, extensions in EXTENSIONS.items():
            if ext in extensions:
                target_folder = os.path.join(directory, folder_name)
                
                # 如果目標資料夾不存在，則建立
                if not os.path.exists(target_folder):
                    os.mkdir(target_folder)
                
                # 移動檔案
                shutil.move(file_path, os.path.join(target_folder, filename))
                print(f"已移動: {filename} -> {folder_name}/")
                moved = True
                break
        
        if not moved:
            # 其他檔案放入 Others
            target_folder = os.path.join(directory, 'Others')
            if not os.path.exists(target_folder):
                os.mkdir(target_folder)
            shutil.move(file_path, os.path.join(target_folder, filename))
            print(f"已移動: {filename} -> Others/")

if __name__ == "__main__":
    target_dir = input("請輸入要整理的資料夾路徑 (預設為當前目錄): ")
    if not target_dir:
        target_dir = "."
    
    organize_files(target_dir)
    print("整理完成！")
```
